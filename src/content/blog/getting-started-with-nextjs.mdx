<h2>Key Features</h2>
<h3>App Router</h3>
<p>The App Router is the recommended way to build applications in Next.js 15:</p>
<pre><code class="language-tsx">// app/page.tsx
export default function Home() {
  return (
    &#x3C;main>
      &#x3C;h1>Welcome to Next.js 15&#x3C;/h1>
    &#x3C;/main>
  );
}
</code></pre>
<h3>Server Components</h3>
<p>Server Components allow you to write components that run on the server:</p>
<pre><code class="language-tsx">// app/blog/page.tsx
async function getPosts() {
  const res = await fetch("https://api.example.com/posts");
  return res.json();
}

export default async function BlogPage() {
  const posts = await getPosts();

  return (
    &#x3C;div>
      {posts.map((post) => (
        &#x3C;article key={post.id}>
          &#x3C;h2>{post.title}&#x3C;/h2>
          &#x3C;p>{post.excerpt}&#x3C;/p>
        &#x3C;/article>
      ))}
    &#x3C;/div>
  );
}
</code></pre>
